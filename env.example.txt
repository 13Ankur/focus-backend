## Backend environment variables (example)
##
## Use this as a reference for your hosting provider's environment config
## (and for local development via a real `.env`, if you use one).
##

# Required (prod)
NODE_ENV=production
PORT=5000
MONGODB_URI=mongodb://localhost:27017/focus
JWT_SECRET=replace-with-a-strong-secret
SENDGRID_API_KEY=replace-with-sendgrid-api-key

# ---- Google Sign-In (ID token verification audiences) ----
# Include ALL client IDs that may produce an ID token:
# - Web client ID (recommended for server-side / offline access)
# - iOS client ID
# - Android client ID
GOOGLE_CLIENT_ID=replace-with-google-web-client-id.apps.googleusercontent.com
GOOGLE_IOS_CLIENT_ID=replace-with-google-ios-client-id.apps.googleusercontent.com
GOOGLE_ANDROID_CLIENT_ID=replace-with-google-android-client-id.apps.googleusercontent.com

# ---- Apple Sign-In (ID token verification audience) ----
# IMPORTANT: Must match the "aud" claim in Apple's identity token.
# For native iOS Sign In with Apple, this is typically your iOS Bundle ID.
APPLE_CLIENT_ID=com.focusapp.buddy

# Optional (only needed if you later exchange authorization codes on the server)
APPLE_TEAM_ID=
APPLE_KEY_ID=
# If storing in env vars, preserve newlines or use \n and convert at runtime.
APPLE_PRIVATE_KEY=
